getar <- function(v){
	vv = cbind(v, c(v[1], v[-length(v)]))
	idx = apply(vv, 1, function(v) prod(!is.nan(v)& v>0))==1
	n = length(which(idx))
	s1 = sum(vv[idx,1]*vv[idx,1])
	s2 = sum(vv[idx,2]*vv[idx,2])
	ar1 = n/(n-1)*sum(vv[idx,1]*vv[idx,2])/sqrt(s1*s2)
	return(ar1)
}
